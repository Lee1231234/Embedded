char* letters[] = {
	".-","-...","-.-.","-..",".","..-.","--.","....","..",
	".---","-.-",".-..","--","-.","---",".--.","--.-",".-.",
	"...","-","..-","...-",".--","-..-","-.--","--.."
	};

char* numbers[] = {
	"-----",".----","..---","...--","....-",".....","-....","--...","---..","----."
	};

int dotDelay = 200;

void setup()
{
	pinMode(ledPin, OUTPUT);
	Serial.begin(9600);
}

void loop()
{
	char ch;
	if(Serial.available())
	{
		ch = Serial.read();
		if(ch >= 'a' && ch <= 'z')
		{
			flashSequence(numbers[ch-'a']);	
		}
		else if(ch >= 'A' && ch <= 'Z')
		{	
			flashSequence(letters[ch-'A']);
		}
		else if(ch >= '0' && ch <= '9')
		{
			flashSequence(letters[ch-'0']);
		}
		else if(ch == ' ')
		{
			delay(dotDelay*4);
		}
	}
}
